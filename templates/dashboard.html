# NexusLeague AI Agent Configuration
# Add these environment variables to your .env or hosting platform:

OPENAI_API_KEY=your_openai_api_key_here
NEXUS_API_KEY=your_league_api_key_here  # Found in your league card above
NEXUS_BASE_URL=https://nexusexporter-clean-production.up.railway.app
NEXUS_LEAGUE_ID=your_league_id_here

# ─── Fetch League Data ───────────────────────────────────────────────
import httpx

BASE = "https://nexusexporter-clean-production.up.railway.app"
API_KEY = "your_league_api_key"
LEAGUE_ID = 1  # Replace with your league ID

def fetch_standings():
    r = httpx.get(f"{BASE}/api/{LEAGUE_ID}/standings", params={"key": API_KEY})
    return r.json()

def fetch_stat_leaders():
    r = httpx.get(f"{BASE}/api/{LEAGUE_ID}/stat_leaders", params={"key": API_KEY})
    return r.json()

def fetch_schedule():
    r = httpx.get(f"{BASE}/api/{LEAGUE_ID}/schedules", params={"key": API_KEY})
    return r.json()

def fetch_rosters():
    r = httpx.get(f"{BASE}/api/{LEAGUE_ID}/rosters", params={"key": API_KEY})
    return r.json()

# ─── Generate AI Headline ────────────────────────────────────────────
from openai import OpenAI

client = OpenAI()

def generate_headline(standings, leaders):
    prompt = f"""
    You are an NFL sports journalist covering a Madden franchise league.

    Current Standings (top teams):
    {standings[:5]}

    Stat Leaders:
    Passing: {leaders.get('pass_yards', [{}])[0]}
    Rushing: {leaders.get('rush_yards', [{}])[0]}

    Write a compelling 2-3 sentence league headline recap.
    """
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}]
    )
    return response.choices[0].message.content

# ─── Example Usage ───────────────────────────────────────────────────
if __name__ == "__main__":
    standings = fetch_standings()
    leaders = fetch_stat_leaders()
    headline = generate_headline(standings, leaders)
    print(headline)
